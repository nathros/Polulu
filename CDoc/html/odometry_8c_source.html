100 void encoder_update(Encoder encoder)
101 {
102  int motor_speed;
103  g_total_count++;
104  if (encoder == LEFT_ENCODER) {
105  motor_speed = motor_read(MOTOR_LEFT);
106  if (motor_speed > 0) {
107  g_left_count++;

109  } else {

111  g_left_rpm_count--;

113  } else {

115  if (motor_speed > 0) {
116  g_right_count++;
117  g_right_rpm_count++;
118  } else {
119  g_right_count--;
120  g_right_rpm_count--;

122  }
